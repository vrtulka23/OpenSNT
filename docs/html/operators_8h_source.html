38template <class A, typename S = EmptySettings>
39class OperatorTernary: public OperatorBase<A, S> {
40public:
41 std::string symbol_other;
42 OperatorTernary(
43 std::string n,
44 std::string s,
45 std::string so,
46 int t
47 ): OperatorBase<A, S>(n, s, t), symbol_other(so) {}
48 virtual void parse(Expression &expr) override {
49 this->groups.clear();
50 expr.remove(this->symbol);
51 bool closed = false;
52 while (expr.right.length()>0) {
53 if (expr.right.rfind(symbol_other, 0) == 0) {
54 expr.remove(symbol_other);
55 this->groups.push_back(expr.pop_left());
56 closed = true;
57 break;
58 }
59 expr.shift();
60 }
61 if (!closed) {
62 throw std::logic_error("Incomplete ternary operator: "+expr.expr);
63 }
64 };

67template <class A, int N=0, typename S = EmptySettings>
68class OperatorGroup: public OperatorBase<A, S> {
69public:

71 std::string symbol_open;
72 std::string symbol_close;
73 std::string symbol_separator;

75 OperatorBase<A, S>(n, s, t), symbol_open("("), symbol_close(")"), symbol_separator(",") {}
76 OperatorGroup(const std::string n, const std::string s, const int t, const std::string so, const std::string sc):
77 OperatorBase<A, S>(n, s, t), symbol_open(so), symbol_close(sc), symbol_separator(",") {}
78 OperatorGroup(const std::string n, const std::string s, const int t, const std::string so, const std::string sc, const std::string ss):
79 OperatorBase<A, S>(n, s, t), symbol_open(so), symbol_close(sc), symbol_separator(ss) {}
80 virtual void parse(Expression &expr) override {
81 this->groups.clear();
82 expr.remove(this->symbol);
83 int depth = 1;
84 while (depth>0) {
85 if (expr.right.length()==0) {
86 throw std::logic_error("Unclosed parentheses: "+expr.expr);
87 } else if (expr.right.rfind(this->symbol, 0) == 0 || expr.right.rfind(symbol_open, 0) == 0) {
88 depth++;
89 } else if (expr.right.rfind(symbol_separator, 0) == 0 && depth==1) {
90 expr.remove(symbol_separator);
91 this->groups.push_back(expr.pop_left());
92 } else if (expr.right.rfind(symbol_close, 0) == 0) {
93 depth--;
94 if (depth==0) {
95 expr.remove(symbol_close);
96 this->groups.push_back(expr.pop_left());
97 break;
98 }
99 }
100 expr.shift();
101 }
102 num_groups = this->groups.size();
103 if (N>0 && num_groups!=N) {
104 throw std::logic_error("Wrong number of group members: "+std::to_string(this->groups.size())+", "+std::to_string(N));
105 }
106 };
