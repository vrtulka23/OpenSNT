38 // Tokenize expression

41 for (auto o : operators.order) {
42 OperatorBase<A,S> *op = operators.select(o);
43 //std::cout << op->symbol << std::endl;

45 is_operator = true;
46 std::string left = expr.pop_left();
47 if (left.length()>0) {
48 tokens.append(ATOM_TOKEN, left);
49 }
50 op->parse(expr);
51 if (op->groups.size()>0) {
52 std::vector<std::string> groups = op->groups;
53 //std::cout << groups.size() << std::endl;
54 for (auto e: groups) {
55 tokens.append(ATOM_TOKEN, solve(e));
56 }
57 }
58 tokens.append(OPERATOR_TOKEN, op->type);
59 //expr.print();
60 }
61 }
62 if (is_operator == false) {
63 expr.shift();
64 //expr.print();
65 }
66 //tokens.print(true);
67 }
68 std::string left = expr.pop_left();
69 if (left.length()>0) {
70 tokens.append(ATOM_TOKEN, left);
71 }
72 //expr.print();
73
74 for (auto s: steps.steps) {
75 //tokens.print(true);
76 tokens.operate(s.second, s.first);
77 }
78 //tokens.print(true);
79
80 if (tokens.left.size()>0 or tokens.right.size()>1) {
81 //tokens.print(true);
82 throw std::logic_error("Cannot solve expression due to unprocessed tokens: "+tokens.to_string(true));
83 }
84
85 Token token = tokens.get_right();

87 return *token.atom;
